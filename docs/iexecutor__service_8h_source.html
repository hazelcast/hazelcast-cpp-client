71  bool cancel(bool may_interrupt_if_running) {
72  if (shared_future_.is_ready()) {
73  return false;
74  }
75 
76  try {
77  return invoke_cancel_request(may_interrupt_if_running);
78  } catch (exception::iexception &) {
79  util::exception_util::rethrow(std::current_exception());
80  }
81  return false;

96  bool invoke_cancel_request(bool may_interrupt_if_running) {
97  invocation_->get_send_connection_or_wait();
98 
99  if (partition_id_ > -1) {
100  auto request = protocol::codec::executorservice_cancelonpartition_encode(uuid_, may_interrupt_if_running);
101  std::shared_ptr<spi::impl::ClientInvocation> clientInvocation = spi::impl::ClientInvocation::create(
102  context_, request, boost::uuids::to_string(uuid_), partition_id_);
103  return clientInvocation->invoke().get().get_first_fixed_sized_field<bool>();
104  } else {
105  auto request = protocol::codec::executorservice_cancelonmember_encode(
106  uuid_, member_uuid_, may_interrupt_if_running);
107  std::shared_ptr<spi::impl::ClientInvocation> clientInvocation = spi::impl::ClientInvocation::create(
108  context_, request, boost::uuids::to_string(uuid_), member_uuid_);
109  return clientInvocation->invoke().get().get_first_fixed_sized_field<bool>();
110  }

496  template<typename T>
497  class MultiExecutionCallbackWrapper : multi_execution_callback<T> {
498  public:
499  MultiExecutionCallbackWrapper(
500  int member_size, const std::shared_ptr<multi_execution_callback<T> > &multi_execution_callback)
501  : multi_execution_callback_(multi_execution_callback), members_(member_size) {
502  }
503 
504  public:
505  void on_response(const member &member, const boost::optional<T> &value) override {
506  multi_execution_callback_->on_response(member, value);
507 
508  std::lock_guard<std::mutex> guard(lock_);
509  values_[member] = value;
510  int waitingResponse = --members_;
511  if (waitingResponse == 0) {
512  on_complete(values_, exceptions_);
513  }
514  }
515 
516  void
517  on_failure(const member &member, std::exception_ptr exception) override {
518  multi_execution_callback_->on_failure(member, exception);
519 
520  std::lock_guard<std::mutex> guard(lock_);
521  exceptions_[member] = exception;
522  int waitingResponse = --members_;
523  if (waitingResponse == 0) {
524  on_complete(values_, exceptions_);
525  }
526  }
527 
528  void on_complete(const std::unordered_map<member, boost::optional<T> > &vals,
529  const std::unordered_map<member, std::exception_ptr> &excs) override {
530  multi_execution_callback_->on_complete(vals, excs);
531  }
532 
533  private:
534 
535  const std::shared_ptr<multi_execution_callback<T> > multi_execution_callback_;
536  // TODO: We may not need thread safe structures here if being used from the same thread
537  std::unordered_map<member, boost::optional<T>> values_;
538  std::unordered_map<member, std::exception_ptr> exceptions_;
539  int members_;
540  std::mutex lock_;
541  };

576  template<typename T>
577  void submit_to_partition_internal(const serialization::pimpl::data &task_data, int partition_id,
578  const std::shared_ptr<execution_callback<T> > &callback) {
579  boost::uuids::uuid uuid = context_.random_uuid();
580 
581  auto messageFuture = invoke_on_partition_internal(task_data, partition_id, uuid);
582 
583  serialization::pimpl::SerializationService *serializationService = &get_serialization_service();
584  spi::impl::ClientExecutionServiceImpl *executionService = &get_context().get_client_execution_service();
585  messageFuture.first.then(boost::launch::sync, [=](boost::future<protocol::ClientMessage> f) {
586  try {
587  auto result = retrieve_result_from_message<T>(serializationService, std::move(f));
588  executionService->execute([=]() { callback->on_response(result); });
589  } catch (exception::iexception &) {
590  auto exception = std::current_exception();
591  executionService->execute([=]() { callback->on_failure(exception); });
592  }
593  });

716  template<typename T>

718  check_sync(

720  boost::uuids::uuid uuid, int partition_id, const member &member, bool prevent_sync) {
721  bool sync = is_sync_computation(prevent_sync);
722  boost::future<boost::optional<T>> objectFuture;
723  if (sync) {
724  objectFuture = retrieve_result_sync<T>(std::move(future_pair.first));
725  } else {
726  serialization::pimpl::SerializationService *serializationService = &get_serialization_service();
727  objectFuture = future_pair.first.then(boost::launch::sync,
728  [=](boost::future<protocol::ClientMessage> f) {
729  return retrieve_result_from_message<T>(
730  serializationService, std::move(f));
731  });
732  }
733 
734  return executor_promise<T>(objectFuture, uuid, partition_id, member.get_uuid(), get_context(), future_pair.second);
735  }
