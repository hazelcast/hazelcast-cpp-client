1486 template<typename T>
1487 generic_record_builder& write(std::string field_name,
1488 T value,
1489 field_kind kind)
1490 {
1491 if (already_built_) {
1492 BOOST_THROW_EXCEPTION(exception::hazelcast_serialization{
1493 boost::str(boost::format("Illegal to write after record is "
1494 "built. {field : %1%, kind : %2%}") %
1495 field_name % kind) });
1496 }
1497
1498 if (strategy_ == strategy::cloner ||
1499 strategy_ == strategy::schema_bounded) {
1500 const auto& schema =
1501 *boost::get<std::shared_ptr<pimpl::schema>>(writer_or_schema_);
1502
1503 check_type_with_schema(schema, field_name, kind);
1504 }
1505
1506 if (strategy_ == strategy::cloner) {
1507 bool inserted = overwritten_fields_.insert(field_name).second;
1508
1509 if (!inserted) {
1510 BOOST_THROW_EXCEPTION(exception::hazelcast_serialization{
1511 "Field can only be written once." });
1512 }
1513 } else if (objects_.find(field_name) != end(objects_)) {
1514 BOOST_THROW_EXCEPTION(exception::hazelcast_serialization{
1515 "Field can only be written once" });
1516 }
1517
1518 if (strategy_ == strategy::cloner) {
1519 auto existing = objects_.find(field_name);
1520
1521 assert(existing != end(objects_));
1522 existing->second = std::move(value);
1523 } else {
1524 (void)objects_.emplace(field_name, std::move(value));
1525 }
1526
1527 if (strategy_ == strategy::default_builder) {
1528 pimpl::schema_writer& writer =
1529 boost::get<pimpl::schema_writer>(writer_or_schema_);
1530
1531 writer.add_field(std::move(field_name), kind);
1532 }
1533
1534 return *this;
1535 }
