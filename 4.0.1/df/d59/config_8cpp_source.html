505  namespace matcher {
506  std::shared_ptr<std::string>
507  matching_point_config_pattern_matcher::matches(const std::vector<std::string> &config_patterns,
508  const std::string &item_name) const {
509  std::shared_ptr<std::string> candidate;
510  std::shared_ptr<std::string> duplicate;
511  int lastMatchingPoint = -1;
512  for (const std::string &pattern : config_patterns) {
513  int matchingPoint = get_matching_point(pattern, item_name);
514  if (matchingPoint > -1 && matchingPoint >= lastMatchingPoint) {
515  if (matchingPoint == lastMatchingPoint) {
516  duplicate = candidate;
517  } else {
518  duplicate.reset();
519  }
520  lastMatchingPoint = matchingPoint;
521  candidate.reset(new std::string(pattern));
522  }
523  }
524  if (duplicate.get() != NULL) {
525  throw (exception::exception_builder<exception::invalid_configuration>(
526  "MatchingPointConfigPatternMatcher::matches") << "Configuration " << item_name
527  << " has duplicate configuration. Candidate:"
528  << *candidate << ", duplicate:"
529  << *duplicate).build();
530  }
531  return candidate;
532  }
533 
534  int matching_point_config_pattern_matcher::get_matching_point(const std::string &pattern,
535  const std::string &item_name) const {
536  size_t index = pattern.find('*');
537  if (index == std::string::npos) {
538  return -1;
539  }
540 
541  std::string firstPart = pattern.substr(0, index);
542  if (item_name.find(firstPart) != 0) {
543  return -1;
544  }
545 
546  std::string secondPart = pattern.substr(index + 1);
547  if (item_name.rfind(secondPart) != (item_name.length() - secondPart.length())) {
548  return -1;
549  }
550 
551  return (int) (firstPart.length() + secondPart.length());
552  }
553  }
