312 template<typename HANDLER>
313 class EntryEventHandler : public HANDLER
314 {
315 public:
316 EntryEventHandler(
317 const std::string& instance_name,
318 spi::impl::ClientClusterServiceImpl& cluster_service,
319 serialization::pimpl::SerializationService& serialization_service,
320 entry_listener&& listener,
321 logger& lg)
322 : instance_name_(instance_name)
323 , cluster_service_(cluster_service)
324 , serialization_service_(serialization_service)
325 , listener_(std::move(listener))
326 , logger_(lg)
327 {}
328
329 void handle_entry(
330 const boost::optional<serialization::pimpl::data>& key,
331 const boost::optional<serialization::pimpl::data>& value,
332 const boost::optional<serialization::pimpl::data>& old_value,
333 const boost::optional<serialization::pimpl::data>& merging_value,
334 int32_t event_type,
335 boost::uuids::uuid uuid,
336 int32_t number_of_affected_entries) override
337 {
338 if (event_type ==
339 static_cast<int32_t>(entry_event::type::CLEAR_ALL)) {
340 fire_map_wide_event(key,
341 value,
342 old_value,
343 merging_value,
344 event_type,
345 uuid,
346 number_of_affected_entries);
347 return;
348 }
349
350 fire_entry_event(key,
351 value,
352 old_value,
353 merging_value,
354 event_type,
355 uuid,
356 number_of_affected_entries);
357 }
358
359 private:
360 void fire_map_wide_event(
361 const boost::optional<serialization::pimpl::data>& key,
362 const boost::optional<serialization::pimpl::data>& value,
363 const boost::optional<serialization::pimpl::data>& old_value,
364 const boost::optional<serialization::pimpl::data>& merging_value,
365 int32_t event_type,
366 boost::uuids::uuid uuid,
367 int32_t number_of_affected_entries)
368 {
369 auto member = cluster_service_.get_member(uuid);
370 auto mapEventType = static_cast<entry_event::type>(event_type);
371 map_event mapEvent(std::move(member).value(),
372 mapEventType,
373 instance_name_,
374 number_of_affected_entries);
375 listener_.map_cleared_(std::move(mapEvent));
376 }
377
378 void fire_entry_event(
379 const boost::optional<serialization::pimpl::data>& key,
380 const boost::optional<serialization::pimpl::data>& value,
381 const boost::optional<serialization::pimpl::data>& old_value,
382 const boost::optional<serialization::pimpl::data>& merging_value,
383 int32_t event_type,
384 boost::uuids::uuid uuid,
385 int32_t number_of_affected_entries)
386 {
387 typed_data eventKey, val, oldVal, mergingVal;
388 if (value) {
389 val = typed_data(std::move(*value), serialization_service_);
390 }
391 if (old_value) {
392 oldVal =
393 typed_data(std::move(*old_value), serialization_service_);
394 }
395 if (merging_value) {
396 mergingVal =
397 typed_data(std::move(*merging_value), serialization_service_);
398 }
399 if (key) {
400 eventKey = typed_data(std::move(*key), serialization_service_);
401 }
402 auto m = cluster_service_.get_member(uuid);
403 if (!m.has_value()) {
404 m = member(uuid);
405 }
406 auto type = static_cast<entry_event::type>(event_type);
407 entry_event entryEvent(instance_name_,
408 std::move(m.value()),
409 type,
410 std::move(eventKey),
411 std::move(val),
412 std::move(oldVal),
413 std::move(mergingVal));
414 switch (type) {
415 case entry_event::type::ADDED:
416 listener_.added_(std::move(entryEvent));
417 break;
418 case entry_event::type::REMOVED:
419 listener_.removed_(std::move(entryEvent));
420 break;
421 case entry_event::type::UPDATED:
422 listener_.updated_(std::move(entryEvent));
423 break;
424 case entry_event::type::EVICTED:
425 listener_.evicted_(std::move(entryEvent));
426 break;
427 default:
428 HZ_LOG(
429 logger_,
430 warning,
431 boost::str(boost::format(
432 "Received unrecognized event with type: %1% "
433 "Dropping the event!!!") %
434 static_cast<int>(type)));
435 }
436 }
437
438 private:
439 const std::string& instance_name_;
440 spi::impl::ClientClusterServiceImpl& cluster_service_;
441 serialization::pimpl::SerializationService& serialization_service_;
442 entry_listener listener_;
443 logger& logger_;
444 };
