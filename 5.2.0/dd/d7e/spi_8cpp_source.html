99void
100ProxyManager::destroy()
101{
102 std::lock_guard<std::recursive_mutex> guard(lock_);
103 for (auto& p : proxies_) {
104 try {
105 auto proxy = p.second.get();
106 p.second.get()->on_shutdown();
107 } catch (std::exception& se) {
108 auto& lg = client_.get_logger();
109 HZ_LOG(
110 lg,
111 finest,
112 boost::str(boost::format(
113 "Proxy was not created, "
114 "hence onShutdown can be called. Exception: %1%") %
115 se.what()));
116 }
117 }
118 proxies_.clear();
119}

135boost::future<void>
136ProxyManager::destroy_proxy(ClientProxy& proxy)
137{
138 DefaultObjectNamespace objectNamespace(proxy.get_service_name(),
139 proxy.get_name());
140 std::shared_ptr<ClientProxy> registeredProxy;
141 {
142 std::lock_guard<std::recursive_mutex> guard(lock_);
143 auto it = proxies_.find(objectNamespace);
144 registeredProxy = it == proxies_.end() ? nullptr : it->second.get();
145 if (it != proxies_.end()) {
146 proxies_.erase(it);
147 }
148 }
149
150 try {
151 if (registeredProxy) {
152 try {
153 proxy.destroy_locally();
154 return proxy.destroy_remotely();
155 } catch (exception::iexception&) {
156 proxy.destroy_remotely();
157 throw;
158 }
159 }
160 if (&proxy != registeredProxy.get()) {
161 // The given proxy is stale and was already destroyed, but the
162 // caller may have allocated local resources in the context of this
163 // stale proxy instance after it was destroyed, so we have to
164 // cleanup it locally one more time to make sure there are no
165 // leaking local resources.
166 proxy.destroy_locally();
167 }
168 } catch (...) {
169 if (&proxy != registeredProxy.get()) {
170 // The given proxy is stale and was already destroyed, but the
171 // caller may have allocated local resources in the context of this
172 // stale proxy instance after it was destroyed, so we have to
173 // cleanup it locally one more time to make sure there are no
174 // leaking local resources.
175 proxy.destroy_locally();
176 }
177 throw;
178 }
179 return boost::make_ready_future();
180}

347bool
348lifecycle_service::start()
349{
350 bool expected = false;
351 if (!active_.compare_exchange_strong(expected, true)) {
352 return false;
353 }
354
355 fire_lifecycle_event(lifecycle_event::STARTED);
356
357 client_context_.get_client_execution_service().start();
358
359 client_context_.get_client_listener_service().start();
360
361 client_context_.get_invocation_service().start();
362
363 client_context_.get_client_cluster_service().start();
364
365 client_context_.get_cluster_view_listener().start();
366
367 if (!client_context_.get_connection_manager().start()) {
368 return false;
369 }
370
371 auto& connectionStrategyConfig =
372 client_context_.get_client_config().get_connection_strategy_config();
373 if (!connectionStrategyConfig.is_async_start()) {
374 // The client needs to open connections to all members before any
375 // services requiring internal listeners start
376 wait_for_initial_membership_event();
377 client_context_.get_connection_manager()
378 .connect_to_all_cluster_members();
379 }
380
381 client_context_.get_invocation_service().add_backup_listener();
382
383 client_context_.get_clientstatistics().start();
384
385 return true;
386}

388void
389lifecycle_service::shutdown()
390{
391 bool expected = true;
392 if (!active_.compare_exchange_strong(expected, false)) {
393 shutdown_completed_latch_.wait();
394 return;
395 }
396 try {
397 fire_lifecycle_event(lifecycle_event::SHUTTING_DOWN);
398 client_context_.get_proxy_session_manager().shutdown();
399 client_context_.get_clientstatistics().shutdown();
400 client_context_.get_proxy_manager().destroy();
401 client_context_.get_connection_manager().shutdown();
402 client_context_.get_client_cluster_service().shutdown();
403 client_context_.get_invocation_service().shutdown();
404 client_context_.get_client_listener_service().shutdown();
405 client_context_.get_near_cache_manager().destroy_all_near_caches();
406 fire_lifecycle_event(lifecycle_event::SHUTDOWN);
407 client_context_.get_client_execution_service().shutdown();
408 client_context_.get_serialization_service().dispose();
409 shutdown_completed_latch_.count_down();
410 } catch (std::exception& e) {
411 HZ_LOG(
412 client_context_.get_logger(),
413 info,
414 boost::str(
415 boost::format(
416 "An exception occured during LifecycleService shutdown. %1%") %
417 e.what()));
418 shutdown_completed_latch_.count_down();
419 }
420}

438void
439lifecycle_service::fire_lifecycle_event(const lifecycle_event& lifecycle_event)
440{
441 std::lock_guard<std::mutex> guard(listener_lock_);
442 logger& lg = client_context_.get_logger();
443
444 std::function<void(lifecycle_listener&)> fire_one;
445
446 switch (lifecycle_event.get_state()) {
447 case lifecycle_event::STARTING: {
448 // convert the date string from "2016-04-20" to 20160420
449 std::string date(HAZELCAST_STRINGIZE(HAZELCAST_GIT_COMMIT_DATE));
450 util::git_date_to_hazelcast_log_date(date);
451 std::string commitId(HAZELCAST_STRINGIZE(HAZELCAST_GIT_COMMIT_ID));
452 commitId.erase(std::remove(commitId.begin(), commitId.end(), '"'),
453 commitId.end());
454
455 HZ_LOG(lg,
456 info,
457 (boost::format("(%1%:%2%) LifecycleService::LifecycleEvent "
458 "Client (%3%) is STARTING") %
459 date % commitId %
460 client_context_.get_connection_manager().get_client_uuid())
461 .str());
462 char msg[100];
463 util::hz_snprintf(
464 msg,
465 100,
466 "(%s:%s) LifecycleService::LifecycleEvent STARTING",
467 date.c_str(),
468 commitId.c_str());
469 HZ_LOG(lg, info, msg);
470
471 fire_one = [](lifecycle_listener& listener) {
472 listener.starting_();
473 };
474 break;
475 }
476 case lifecycle_event::STARTED: {
477 HZ_LOG(lg, info, "LifecycleService::LifecycleEvent STARTED");
478
479 fire_one = [](lifecycle_listener& listener) {
480 listener.started_();
481 };
482 break;
483 }
484 case lifecycle_event::SHUTTING_DOWN: {
485 HZ_LOG(lg, info, "LifecycleService::LifecycleEvent SHUTTING_DOWN");
486
487 fire_one = [](lifecycle_listener& listener) {
488 listener.shutting_down_();
489 };
490 break;
491 }
492 case lifecycle_event::SHUTDOWN: {
493 HZ_LOG(lg, info, "LifecycleService::LifecycleEvent SHUTDOWN");
494
495 fire_one = [](lifecycle_listener& listener) {
496 listener.shutdown_();
497 };
498 break;
499 }
500 case lifecycle_event::CLIENT_CONNECTED: {
501 HZ_LOG(
502 lg, info, "LifecycleService::LifecycleEvent CLIENT_CONNECTED");
503
504 fire_one = [](lifecycle_listener& listener) {
505 listener.connected_();
506 };
507 break;
508 }
509 case lifecycle_event::CLIENT_DISCONNECTED: {
510 HZ_LOG(
511 lg, info, "LifecycleService::LifecycleEvent CLIENT_DISCONNECTED");
512
513 fire_one = [](lifecycle_listener& listener) {
514 listener.disconnected_();
515 };
516 break;
517 }
518 }
519
520 for (auto& item : listeners_) {
521 fire_one(item.second);
522 }
523}

752void

754 const std::shared_ptr<ClientInvocation>& invocation,
755 const std::shared_ptr<protocol::ClientMessage>& response)
756{
757 try {
758 if (protocol::codec::ErrorCodec::EXCEPTION_MESSAGE_TYPE ==
759 response->get_message_type()) {
760 auto error_holder = protocol::codec::ErrorCodec::decode(*response);
761 invocation->notify_exception(
762 client_.get_client_exception_factory().create_exception(
763 error_holder));
764 } else {
765 invocation->notify(response);
766 }
767 } catch (std::exception& e) {
768 HZ_LOG(

771 boost::str(boost::format("Failed to process response for %1%. %2%") %
772 *invocation % e.what()));

816 // If the client is initialized on the cluster, that means we
817 // have sent all the schemas to the cluster, even if we are
818 // reconnected to it
819 return;
820 }
821
822 if (!client_.get_hazelcast_client_implementation()
823 ->should_check_urgent_invocations()) {
824 // If there were no Compact schemas to begin with, we don't need
825 // to perform the check below. If the client didn't send a Compact
826 // schema up until this point, the retries or listener registrations
827 // could not send a schema, because if they were, we wouldn't hit
828 // this line.
829 return;
830 }
831
832 // We are not yet initialized on cluster, so the Compact schemas might
833 // not be sent yet. This message contains some serialized classes,
834 // and it is possible that it can also contain Compact serialized data.
835 // In that case, allowing this invocation to go through now could
836 // violate the invariant that the schema must come to cluster before
837 // the data. We will retry this invocation and wait until the client
838 // is initialized on the cluster, which means schemas are replicated
839 // in the cluster.
840 if (invocation.get_client_message()
841 ->contains_serialized_data_in_request()) {
842 throw exception::invocation_might_contain_compact_data{
843 "ClientInvocationServiceImpl::check_urgent_invocation_allowed",
844 invocation
845 };
846 }
847}

1074void
1075ClientClusterServiceImpl::handle_event(int32_t version,
1076 const std::vector<member>& member_infos)
1077{
1078 auto& lg = client_.get_logger();
1079 HZ_LOG(
1080 lg,
1081 finest,
1082 boost::str(
1083 boost::format("Handling new snapshot with membership version: %1%, "
1084 "membersString %2%") %
1085 version % members_string(create_snapshot(version, member_infos))));
1086 auto cluster_view_snapshot = member_list_snapshot_.load();
1087 if (cluster_view_snapshot == EMPTY_SNAPSHOT) {
1088 std::lock_guard<std::mutex> g(cluster_view_lock_);
1089 cluster_view_snapshot = member_list_snapshot_.load();
1090 if (cluster_view_snapshot == EMPTY_SNAPSHOT) {
1091 // this means this is the first time client connected to cluster
1092 apply_initial_state(version, member_infos);
1093 initial_list_fetched_latch_.count_down();
1094 return;
1095 }
1096 }
1097
1098 std::vector<membership_event> events;
1099 if (version >= cluster_view_snapshot->version) {
1100 std::lock_guard<std::mutex> g(cluster_view_lock_);
1101 cluster_view_snapshot = member_list_snapshot_.load();
1102 if (version >= cluster_view_snapshot->version) {
1103 auto prev_members = cluster_view_snapshot->members;
1104 auto snapshot = boost::make_shared<member_list_snapshot>(
1105 create_snapshot(version, member_infos));
1106 member_list_snapshot_.store(snapshot);
1107 events = detect_membership_events(prev_members, snapshot->members);
1108 }
1109 }
1110
1111 fire_events(std::move(events));
1112}
1113
1114ClientClusterServiceImpl::member_list_snapshot
1115ClientClusterServiceImpl::create_snapshot(int32_t version,
1116 const std::vector<member>& members)
1117{
1118 member_list_snapshot result;
1119 result.version = version;
1120 for (auto& m : members) {
1121 auto const& address_map = m.address_map();
1122 if (address_map.empty()) {
1123 result.members.insert({ m.get_uuid(), m });
1124 } else {
1125 auto found = address_map.find(CLIENT);
1126 address member_address;
1127 if (found != address_map.end()) {
1128 member_address = found->second;
1129 } else {
1130 found = address_map.find(MEMBER);
1131 assert(found != address_map.end());
1132 member_address = found->second;
1133 }
1134 member new_member(member_address,
1135 m.get_uuid(),
1136 m.is_lite_member(),
1137 m.get_attributes(),
1138 m.address_map(),
1139 m.get_version());
1140 result.members.emplace(new_member.get_uuid(),
1141 std::move(new_member));
1142 }
1143 }
1144
1145 return result;
1146}

1162void
1163ClientClusterServiceImpl::apply_initial_state(
1164 int32_t version,
1165 const std::vector<member>& member_infos)
1166{

1169 member_list_snapshot_.store(snapshot);

1171 std::unordered_set<member> members;
1172 for (auto const& e : snapshot->members) {
1173 members.insert(e.second);
1174 }
1175 std::lock_guard<std::mutex> g(listeners_lock_);
1176 for (auto& item : listeners_) {
1177 membership_listener& listener = item.second;
1178 if (listener.init_) {

1185std::vector<membership_event>
1186ClientClusterServiceImpl::detect_membership_events(
1187 std::unordered_map<boost::uuids::uuid,
1188 member,
1189 boost::hash<boost::uuids::uuid>> previous_members,
1190 const std::unordered_map<boost::uuids::uuid,
1191 member,
1192 boost::hash<boost::uuids::uuid>>& current_members)
1193{
1194 std::vector<member> new_members;
1195
1196 for (auto const& e : current_members) {
1197 if (!previous_members.erase(e.first)) {
1198 new_members.emplace_back(e.second);
1199 }
1200 }
1201
1202 std::vector<membership_event> events;
1203
1204 // removal events should be added before added events
1205 for (auto const& e : previous_members) {
1206 events.emplace_back(
1207 client_.get_cluster(),
1208 e.second,
1209 membership_event::membership_event_type::MEMBER_LEFT,
1210 current_members);
1211 auto connection =
1212 client_.get_connection_manager().get_connection(e.second.get_uuid());
1213 if (connection) {
1214 connection->close(
1215 "",
1216 std::make_exception_ptr(exception::target_disconnected(
1217 "ClientClusterServiceImpl::detect_membership_events",
1218 (boost::format(
1219 "The client has closed the connection to this member, after "
1220 "receiving a member left event from the cluster. %1%") %
1221 *connection)
1222 .str())));
1223 }
1224 }
1225 for (auto const& member : new_members) {
1226 events.emplace_back(
1227 client_.get_cluster(),
1228 member,
1229 membership_event::membership_event_type::MEMBER_JOINED,
1230 current_members);
1231 }
1232
1233 if (!events.empty()) {
1234 auto snapshot = member_list_snapshot_.load();
1235 if (!snapshot->members.empty()) {
1236 HZ_LOG(client_.get_logger(), info, members_string(*snapshot));
1237 }
1238 }
1239 return events;
1240}

1244{
1245 std::lock_guard<std::mutex> g(listeners_lock_);
1246
1247 for (auto const& event : events) {
1248 for (auto& item : listeners_) {
1249 membership_listener& listener = item.second;
1250 if (event.get_event_type() ==
1251 membership_event::membership_event_type::MEMBER_JOINED) {
1252 listener.joined_(event);
1253 } else {
1254 listener.left_(event);
1255 }
1256 }
1257 }

1435boost::future<protocol::ClientMessage>
1436ClientInvocation::invoke()
1437{
1438 assert(client_message_.load());
1439
1440 auto actual_work = [this]() {
1441 // for back pressure
1442 call_id_sequence_->next();
1443 invoke_on_selection();
1444 if (!lifecycle_service_.is_running()) {
1445 return invocation_promise_.get_future().then(
1446 [](boost::future<protocol::ClientMessage> f) { return f.get(); });
1447 }
1448 auto id_seq = call_id_sequence_;
1449 return invocation_promise_.get_future().then(
1450 execution_service_->get_user_executor(),
1451 [=](boost::future<protocol::ClientMessage> f) {
1452 id_seq->complete();
1453 return f.get();
1454 });
1455 };
1456
1457 const auto& schemas =
1458 (*(client_message_.load()))->schemas_will_be_replicated();
1459
1460 if (!schemas.empty()) {
1461 auto self = shared_from_this();
1462
1463 return replicate_schemas(schemas)
1464 .then(boost::launch::sync,
1465 [actual_work, self](boost::future<void> replication) {
1466 replication.get();
1467
1468 return actual_work();
1469 })
1470 .unwrap();
1471 }
1472
1473 return actual_work();
1474}

1517void
1518ClientInvocation::invoke_on_selection()
1519{
1520 try {
1521 invoke_count_++;
1522 if (urgent_) {
1523 invocation_service_.check_urgent_invocation_allowed(*this);
1524 } else {
1525 invocation_service_.check_invocation_allowed();
1526 }
1527
1528 if (is_bind_to_single_connection()) {
1529 bool invoked = false;
1530 auto conn = connection_.lock();
1531 if (conn) {
1532 invoked = invocation_service_.invoke_on_connection(
1533 shared_from_this(), conn);
1534 }
1535 if (!invoked) {
1536 std::string message;
1537 if (conn) {
1538 message =
1539 (boost::format("Could not invoke on connection %1%") %
1540 *conn)
1541 .str();
1542 } else {
1543 message = "Could not invoke. Bound to a connection that is "
1544 "deleted already.";
1545 }

1562 }
1563 if (!invoked) {
1564 invoked = invocation_service_.invoke(shared_from_this());
1565 }
1566 } else {
1567 invoked = invocation_service_.invoke(shared_from_this());
1568 }
1569 if (!invoked) {

1906void
1907ClientInvocation::notify(const std::shared_ptr<protocol::ClientMessage>& msg)
1908{
1909 if (!msg) {
1910 BOOST_THROW_EXCEPTION(
1911 exception::illegal_argument("response can't be null"));
1912 }
1913
1914 int8_t expected_backups = msg->get_number_of_backups();
1915
1916 // if a regular response comes and there are backups, we need to wait for
1917 // the backups when the backups complete, the response will be send by the
1918 // last backup or backup-timeout-handle mechanism kicks on
1919 if (expected_backups > backup_acks_received_) {
1920 // so the invocation has backups and since not all backups have
1921 // completed, we need to wait (it could be that backups arrive earlier
1922 // than the response)
1923
1924 pending_response_received_time_ = std::chrono::steady_clock::now();
1925
1926 backup_acks_expected_ = expected_backups;
1927
1928 // it is very important that the response is set after the
1929 // backupsAcksExpected is set, else the system can assume the invocation
1930 // is complete because there is a response and no backups need to
1931 // respond
1932 pending_response_ = msg;
1933
1934 // we are done since not all backups have completed. Therefore we should
1935 // not notify the future
1936 return;
1937 }
1938
1939 // we are going to notify the future that a response is available; this can
1940 // happen when:
1941 // - we had a regular operation (so no backups we need to wait for) that
1942 // completed
1943 // - we had a backup-aware operation that has completed, but also all its
1944 // backups have completed
1945 complete(msg);
1946}

1984void
1985ClientInvocation::execute()
1986{
1987 auto this_invocation = shared_from_this();
1988 auto command = [=]() { this_invocation->run(); };
1989
1990 // first we force a new invocation slot because we are going to return our
1991 // old invocation slot immediately after It is important that we first
1992 // 'force' taking a new slot; otherwise it could be that a sneaky invocation
1993 // gets through that takes our slot!
1994 int64_t callId = call_id_sequence_->force_next();
1995 client_message_.load()->get()->set_correlation_id(callId);
1996
1997 // we release the old slot
1998 call_id_sequence_->complete();
1999
2000 if (invoke_count_ < MAX_FAST_INVOCATION_COUNT) {
2001 // fast retry for the first few invocations
2002 execution_service_->execute(command);
2003 } else {
2004 // progressive retry delay
2005 int64_t delayMillis = util::min<int64_t>(
2006 static_cast<int64_t>(1)
2007 << (invoke_count_ - MAX_FAST_INVOCATION_COUNT),
2008 std::chrono::duration_cast<std::chrono::milliseconds>(retry_pause_)
2009 .count());
2010 retry_timer_ = execution_service_->schedule(
2011 command, std::chrono::milliseconds(delayMillis));
2012 }
2013}

2044void
2045ClientInvocation::notify_backup()
2046{
2047 ++backup_acks_received_;
2048
2049 if (!pending_response_) {
2050 // no pendingResponse has been set, so we are done since the invocation
2051 // on the primary needs to complete first
2052 return;
2053 }
2054
2055 // if a pendingResponse is set, then the backupsAcksExpected has been set
2056 // (so we can now safely read backupsAcksExpected)
2057 if (backup_acks_expected_ != backup_acks_received_) {
2058 // we managed to complete a backup, but we were not the one completing
2059 // the last backup, so we are done
2060 return;
2061 }
2062
2063 // we are the lucky one since we just managed to complete the last backup
2064 // for this invocation and since the pendingResponse is set, we can set it
2065 // on the future
2066 complete_with_pending_response();
2067}

2069void
2070ClientInvocation::detect_and_handle_backup_timeout(
2071 const std::chrono::milliseconds& backup_timeout)
2072{
2073 // if the backups have completed, we are done; this also filters out all non
2074 // backup-aware operations since the backupsAcksExpected will always be
2075 // equal to the backupsAcksReceived
2076 if (backup_acks_expected_ == backup_acks_received_) {
2077 return;
2078 }
2079
2080 // if no response has yet been received, we we are done; we are only going
2081 // to re-invoke an operation if the response of the primary has been
2082 // received, but the backups have not replied
2083 if (!pending_response_) {
2084 return;
2085 }
2086
2087 // if this has not yet expired (so has not been in the system for a too long
2088 // period) we ignore it
2089 if (pending_response_received_time_ + backup_timeout >=
2090 std::chrono::steady_clock::now()) {
2091 return;
2092 }
2093
2094 if (invocation_service_.fail_on_indeterminate_state()) {
2095 auto exception = boost::enable_current_exception(
2096 (exception::exception_builder<
2097 exception::indeterminate_operation_state>(
2098 "ClientInvocation::detect_and_handle_backup_timeout")
2099 << *this << " failed because backup acks missed.")
2100 .build());
2101 notify_exception(std::make_exception_ptr(exception));
2102 return;
2103 }
2104
2105 // the backups have not yet completed, but we are going to release the
2106 // future anyway if a pendingResponse has been set
2107 complete_with_pending_response();
2108}
