102  void ProxyManager::destroy() {
103  std::lock_guard<std::recursive_mutex> guard(lock_);
104  for (auto &p : proxies_) {
105  try {
106  auto proxy = p.second.get();
107  p.second.get()->on_shutdown();
108  } catch (std::exception &se) {
109  auto &lg = client_.get_logger();
110  HZ_LOG(lg, finest,
111  boost::str(boost::format("Proxy was not created, "
112  "hence onShutdown can be called. Exception: %1%")
113  % se.what())
114  );
115  }
116  }
117  proxies_.clear();
118  }

131  boost::future<void> ProxyManager::destroy_proxy(ClientProxy &proxy) {
132  DefaultObjectNamespace objectNamespace(proxy.get_service_name(), proxy.get_name());
133  std::shared_ptr<ClientProxy> registeredProxy;
134  {
135  std::lock_guard<std::recursive_mutex> guard(lock_);
136  auto it = proxies_.find(objectNamespace);
137  registeredProxy = it == proxies_.end() ? nullptr : it->second.get();
138  if (it != proxies_.end()) {
139  proxies_.erase(it);
140  }
141  }
142 
143  try {
144  if (registeredProxy) {
145  try {
146  proxy.destroy_locally();
147  return proxy.destroy_remotely();
148  } catch (exception::iexception &) {
149  proxy.destroy_remotely();
150  throw;
151  }
152  }
153  if (&proxy != registeredProxy.get()) {
154  // The given proxy is stale and was already destroyed, but the caller
155  // may have allocated local resources in the context of this stale proxy
156  // instance after it was destroyed, so we have to cleanup it locally one
157  // more time to make sure there are no leaking local resources.
158  proxy.destroy_locally();
159  }
160  } catch (...) {
161  if (&proxy != registeredProxy.get()) {
162  // The given proxy is stale and was already destroyed, but the caller
163  // may have allocated local resources in the context of this stale proxy
164  // instance after it was destroyed, so we have to cleanup it locally one
165  // more time to make sure there are no leaking local resources.
166  proxy.destroy_locally();
167  }
168  throw;
169  }
170  return boost::make_ready_future();
171  }

284  bool lifecycle_service::start() {
285  bool expected = false;
286  if (!active_.compare_exchange_strong(expected, true)) {
287  return false;
288  }
289 
290  fire_lifecycle_event(lifecycle_event::STARTED);
291 
292  client_context_.get_client_execution_service().start();
293 
294  client_context_.get_client_listener_service().start();
295 
296  client_context_.get_invocation_service().start();
297 
298  client_context_.get_client_cluster_service().start();
299 
300  client_context_.get_cluster_view_listener().start();
301 
302  if (!client_context_.get_connection_manager().start()) {
303  return false;
304  }
305 
306  auto &connectionStrategyConfig = client_context_.get_client_config().get_connection_strategy_config();
307  if (!connectionStrategyConfig.is_async_start()) {
308  // The client needs to open connections to all members before any services requiring internal listeners start
309  wait_for_initial_membership_event();
310  client_context_.get_connection_manager().connect_to_all_cluster_members();
311  }
312 
313  client_context_.get_invocation_service().add_backup_listener();
314 
315  client_context_.get_clientstatistics().start();
316 
317  return true;
318  }

320  void lifecycle_service::shutdown() {
321  bool expected = true;
322  if (!active_.compare_exchange_strong(expected, false)) {
323  shutdown_completed_latch_.wait();
324  return;
325  }
326  try {
327  fire_lifecycle_event(lifecycle_event::SHUTTING_DOWN);
328  client_context_.get_proxy_session_manager().shutdown();
329  client_context_.get_clientstatistics().shutdown();
330  client_context_.get_proxy_manager().destroy();
331  client_context_.get_connection_manager().shutdown();
332  client_context_.get_client_cluster_service().shutdown();
333  client_context_.get_invocation_service().shutdown();
334  client_context_.get_client_listener_service().shutdown();
335  client_context_.get_near_cache_manager().destroy_all_near_caches();
336  fire_lifecycle_event(lifecycle_event::SHUTDOWN);
337  client_context_.get_client_execution_service().shutdown();
338  client_context_.get_serialization_service().dispose();
339  shutdown_completed_latch_.count_down();
340  } catch (std::exception &e) {
341  HZ_LOG(client_context_.get_logger(), info,
342  boost::str(boost::format("An exception occured during LifecycleService shutdown. %1%")
343  % e.what())
344  );
345  shutdown_completed_latch_.count_down();
346  }
347  }

361  void lifecycle_service::fire_lifecycle_event(const lifecycle_event &lifecycle_event) {
362  std::lock_guard<std::mutex> guard(listener_lock_);
363  logger &lg = client_context_.get_logger();
364 
365  std::function<void(lifecycle_listener &)> fire_one;
366 
367  switch (lifecycle_event.get_state()) {
368  case lifecycle_event::STARTING : {
369  // convert the date string from "2016-04-20" to 20160420
370  std::string date(HAZELCAST_STRINGIZE(HAZELCAST_GIT_COMMIT_DATE));
371  util::git_date_to_hazelcast_log_date(date);
372  std::string commitId(HAZELCAST_STRINGIZE(HAZELCAST_GIT_COMMIT_ID));
373  commitId.erase(std::remove(commitId.begin(), commitId.end(), '"'), commitId.end());
374 
375  HZ_LOG(lg, info,
376  (boost::format("(%1%:%2%) LifecycleService::LifecycleEvent Client (%3%) is STARTING") %
377  date % commitId % client_context_.get_connection_manager().get_client_uuid()).str());
378  char msg[100];
379  util::hz_snprintf(msg, 100, "(%s:%s) LifecycleService::LifecycleEvent STARTING", date.c_str(),
380  commitId.c_str());
381  HZ_LOG(lg, info, msg);
382 
383  fire_one = [](lifecycle_listener &listener) {
384  listener.starting_();
385  };
386  break;
387  }
388  case lifecycle_event::STARTED : {
389  HZ_LOG(lg, info, "LifecycleService::LifecycleEvent STARTED");
390 
391  fire_one = [](lifecycle_listener &listener) {
392  listener.started_();
393  };
394  break;
395  }
396  case lifecycle_event::SHUTTING_DOWN : {
397  HZ_LOG(lg, info, "LifecycleService::LifecycleEvent SHUTTING_DOWN");
398 
399  fire_one = [](lifecycle_listener &listener) {
400  listener.shutting_down_();
401  };
402  break;
403  }
404  case lifecycle_event::SHUTDOWN : {
405  HZ_LOG(lg, info, "LifecycleService::LifecycleEvent SHUTDOWN");
406 
407  fire_one = [](lifecycle_listener &listener) {
408  listener.shutdown_();
409  };
410  break;
411  }
412  case lifecycle_event::CLIENT_CONNECTED : {
413  HZ_LOG(lg, info, "LifecycleService::LifecycleEvent CLIENT_CONNECTED");
414 
415  fire_one = [](lifecycle_listener &listener) {
416  listener.connected_();
417  };
418  break;
419  }
420  case lifecycle_event::CLIENT_DISCONNECTED : {
421  HZ_LOG(lg, info, "LifecycleService::LifecycleEvent CLIENT_DISCONNECTED");
422 
423  fire_one = [](lifecycle_listener &listener) {
424  listener.disconnected_();
425  };
426  break;
427  }
428  }
429 
430  for (auto &item: listeners_) {
431  fire_one(item.second);
432  }
433  }

844  ClientClusterServiceImpl::member_list_snapshot
845  ClientClusterServiceImpl::create_snapshot(int32_t version, const std::vector<member> &members) {
846  member_list_snapshot result;
847  result.version = version;
848  for (auto &m : members) {
849  auto const &address_map = m.address_map();
850  if (address_map.empty()) {
851  result.members.insert({m.get_uuid(), m});
852  } else {
853  auto found = address_map.find(CLIENT);
854  address member_address;
855  if (found != address_map.end()) {
856  member_address = found->second;
857  } else {
858  found = address_map.find(MEMBER);
859  assert(found != address_map.end());
860  member_address = found->second;
861  }
862  member new_member(member_address, m.get_uuid(), m.is_lite_member(), m.get_attributes(), m.address_map());
863  result.members.emplace(new_member.get_uuid(), std::move(new_member));
864  }
865  }
866 
867  return result;
868  }

900  std::vector<membership_event> ClientClusterServiceImpl::detect_membership_events(
901  std::unordered_map<boost::uuids::uuid, member, boost::hash<boost::uuids::uuid>> previous_members,
902  const std::unordered_map<boost::uuids::uuid, member, boost::hash<boost::uuids::uuid>>& current_members) {
903  std::vector<member> new_members;
904 
905  for (auto const & e : current_members) {
906  if (!previous_members.erase(e.first)) {
907  new_members.emplace_back(e.second);
908  }
909  }
910 
911  std::vector<membership_event> events;
912 
913  // removal events should be added before added events
914  for (auto const &e : previous_members) {
915  events.emplace_back(client_.get_cluster(), e.second, membership_event::membership_event_type::MEMBER_LEFT, current_members);
916  auto connection = client_.get_connection_manager().get_connection(e.second.get_uuid());
917  if (connection) {
918  connection->close("", std::make_exception_ptr(exception::target_disconnected(
919  "ClientClusterServiceImpl::detect_membership_events", (boost::format(
920  "The client has closed the connection to this member, after receiving a member left event from the cluster. %1%") %
921  *connection).str())));
922  }
923  }
924  for (auto const &member : new_members) {
925  events.emplace_back(client_.get_cluster(), member, membership_event::membership_event_type::MEMBER_JOINED, current_members);
926  }
927 
928  if (!events.empty()) {
929  auto snapshot = member_list_snapshot_.load();
930  if (!snapshot->members.empty()) {
931  HZ_LOG(client_.get_logger(), info, members_string(*snapshot));
932  }
933  }
934  return events;
935  }

1204  void ClientInvocation::notify_exception(std::exception_ptr exception) {
1205  erase_invocation();
1206  try {
1207  std::rethrow_exception(exception);
1208  } catch (exception::iexception &iex) {
1209  log_exception(iex);
1210 
1211  if (!lifecycle_service_.is_running()) {
1212  try {
1213  std::throw_with_nested(boost::enable_current_exception(
1214  exception::hazelcast_client_not_active(iex.get_source(),
1215  "Client is shutting down")));
1216  } catch (exception::iexception &e) {
1217  set_exception(e, boost::current_exception());
1218  }
1219  return;
1220  }
1221 
1222  if (!should_retry(iex)) {
1223  set_exception(iex, boost::current_exception());
1224  return;
1225  }
1226 
1227  auto timePassed = std::chrono::steady_clock::now() - start_time_;
1228  if (timePassed > invocation_service_.get_invocation_timeout()) {
1229  HZ_LOG(logger_, finest,
1230  boost::str(boost::format("Exception will not be retried because "
1231  "invocation timed out. %1%") % iex.what())
1232  );
1233

1402  void ClientInvocation::notify(const std::shared_ptr<protocol::ClientMessage> &msg) {
1403  if (!msg) {
1404  BOOST_THROW_EXCEPTION(exception::illegal_argument("response can't be null"));
1405  }
1406 
1407  int8_t expected_backups = msg->get_number_of_backups();
1408 
1409  // if a regular response comes and there are backups, we need to wait for the backups
1410  // when the backups complete, the response will be send by the last backup or backup-timeout-handle mechanism kicks on
1411  if (expected_backups > backup_acks_received_) {
1412  // so the invocation has backups and since not all backups have completed, we need to wait
1413  // (it could be that backups arrive earlier than the response)
1414 
1415  pending_response_received_time_ = std::chrono::steady_clock::now();
1416 
1417  backup_acks_expected_ = expected_backups;
1418 
1419  // it is very important that the response is set after the backupsAcksExpected is set, else the system
1420  // can assume the invocation is complete because there is a response and no backups need to respond
1421  pending_response_ = msg;
1422 
1423  // we are done since not all backups have completed. Therefore we should not notify the future
1424  return;
1425  }
1426 
1427  // we are going to notify the future that a response is available; this can happen when:
1428  // - we had a regular operation (so no backups we need to wait for) that completed
1429  // - we had a backup-aware operation that has completed, but also all its backups have completed
1430  complete(msg);
1431  }

1461  void ClientInvocation::execute() {
1462  auto this_invocation = shared_from_this();
1463  auto command = [=]() {
1464  this_invocation->run();
1465  };
1466 
1467  // first we force a new invocation slot because we are going to return our old invocation slot immediately after
1468  // It is important that we first 'force' taking a new slot; otherwise it could be that a sneaky invocation gets
1469  // through that takes our slot!
1470  int64_t callId = call_id_sequence_->force_next();
1471  client_message_.load()->get()->set_correlation_id(callId);
1472 
1473  //we release the old slot
1474  call_id_sequence_->complete();
1475 
1476  if (invoke_count_ < MAX_FAST_INVOCATION_COUNT) {
1477  // fast retry for the first few invocations
1478  execution_service_->execute(command);
1479  } else {
1480  // progressive retry delay
1481  int64_t delayMillis = util::min<int64_t>(static_cast<int64_t>(1) << (invoke_count_ - MAX_FAST_INVOCATION_COUNT),
1482  std::chrono::duration_cast<std::chrono::milliseconds>(retry_pause_).count());
1483  retry_timer_ = execution_service_->schedule(command, std::chrono::milliseconds(delayMillis));
1484  }
1485  }

1507  void ClientInvocation::notify_backup() {
1508  ++backup_acks_received_;
1509 
1510  if (!pending_response_) {
1511  // no pendingResponse has been set, so we are done since the invocation on the primary needs to complete first
1512  return;
1513  }
1514 
1515  // if a pendingResponse is set, then the backupsAcksExpected has been set (so we can now safely read backupsAcksExpected)
1516  if (backup_acks_expected_ != backup_acks_received_) {
1517  // we managed to complete a backup, but we were not the one completing the last backup, so we are done
1518  return;
1519  }
1520 
1521  // we are the lucky one since we just managed to complete the last backup for this invocation and since the
1522  // pendingResponse is set, we can set it on the future
1523  complete_with_pending_response();
1524  }

1526  void
1527  ClientInvocation::detect_and_handle_backup_timeout(const std::chrono::milliseconds &backup_timeout) {
1528  // if the backups have completed, we are done; this also filters out all non backup-aware operations
1529  // since the backupsAcksExpected will always be equal to the backupsAcksReceived
1530  if (backup_acks_expected_ == backup_acks_received_) {
1531  return;
1532  }
1533 
1534  // if no response has yet been received, we we are done; we are only going to re-invoke an operation
1535  // if the response of the primary has been received, but the backups have not replied
1536  if (!pending_response_) {
1537  return;
1538  }
1539 
1540  // if this has not yet expired (so has not been in the system for a too long period) we ignore it
1541  if (pending_response_received_time_ + backup_timeout >= std::chrono::steady_clock::now()) {
1542  return;
1543  }
1544 
1545  if (invocation_service_.fail_on_indeterminate_state()) {
1546  auto exception = boost::enable_current_exception((exception::exception_builder<exception::indeterminate_operation_state>(
1547  "ClientInvocation::detect_and_handle_backup_timeout") << *this
1548  << " failed because backup acks missed.").build());
1549  notify_exception(std::make_exception_ptr(exception));
1550  return;
1551  }
1552 
1553  // the backups have not yet completed, but we are going to release the future anyway if a pendingResponse has been set
1554  complete_with_pending_response();
1555  }
