658 template<typename K, typename V>
659 boost::future<std::unordered_map<K, V>> get_all(
660 const std::unordered_set<K>& keys)
661 {
662 if (keys.empty()) {
663 return boost::make_ready_future(std::unordered_map<K, V>());
664 }
665
666 std::unordered_map<int, std::vector<serialization::pimpl::data>>
667 partition_to_key_data;
668 // group the request per server
669 for (auto& key : keys) {
670 auto key_data = to_data<K>(key);
671
672 auto partitionId = get_partition_id(key_data);
673 partition_to_key_data[partitionId].push_back(std::move(key_data));
674 }
675
676 std::vector<boost::future<EntryVector>> futures;
677 futures.reserve(partition_to_key_data.size());
678 for (auto& entry : partition_to_key_data) {
679 futures.push_back(get_all_internal(entry.first, entry.second));
680 }
681
682 return to_object_map<K, V>(futures);
683 }

1045 template<typename K, typename V>
1046 boost::future<void> put_all(const std::unordered_map<K, V>& entries)
1047 {
1048 std::unordered_map<int, EntryVector> entryMap;
1049 for (auto& entry : entries) {
1050 serialization::pimpl::data key_data = to_data(entry.first);
1051 int partitionId = get_partition_id(key_data);
1052 entryMap[partitionId].push_back(
1053 std::make_pair(key_data, to_data(entry.second)));
1054 }
1055
1056 std::vector<boost::future<protocol::ClientMessage>> resultFutures;
1057 for (auto&& partitionEntry : entryMap) {
1058 auto partitionId = partitionEntry.first;
1059 resultFutures.push_back(
1060 put_all_internal(partitionId, std::move(partitionEntry.second)));
1061 }
1062 return boost::when_all(resultFutures.begin(), resultFutures.end())
1063 .then(boost::launch::sync,
1064 [](boost::future<boost::csbl::vector<
1065 boost::future<protocol::ClientMessage>>> futures) {
1066 for (auto& f : futures.get()) {
1067 f.get();
1068 }
1069 });
1070 }

1301 template<typename K, typename V>
1302 std::vector<std::pair<K, boost::optional<V>>> sort_and_get(
1303 query::paging_predicate<K, V>& predicate,
1304 query::iteration_type iteration_type,
1305 std::vector<std::pair<K, boost::optional<V>>> entries)
1306 {
1307 std::sort(entries.begin(),
1308 entries.end(),
1309 [&](const std::pair<K, boost::optional<V>>& lhs,
1310 const std::pair<K, boost::optional<V>>& rhs) {
1311 auto comparator = predicate.getComparator();
1312 if (!comparator) {
1313 switch (predicate.getIterationType()) {
1314 case query::iteration_type::VALUE:
1315 return lhs.second < rhs.second;
1316 default:
1317 return lhs.first < rhs.first;
1318 }
1319 }
1320
1321 std::pair<const K*, const V*> leftVal(
1322 &lhs.first, lhs.second.get_ptr());
1323 std::pair<const K*, const V*> rightVal(
1324 &rhs.first, rhs.second.get_ptr());
1325 int result = comparator->compare(&leftVal, &rightVal);
1326 if (0 != result) {
1327 // std sort: comparison function object returns
1328 // true if the first argument is less than (i.e. is
1329 // ordered before) the second.
1330 return result < 0;
1331 }
1332
1333 return lhs.first < rhs.first;
1334 });
1335
1336 std::pair<size_t, size_t> range =
1337 update_anchor<K, V>(entries, predicate, iteration_type);
1338
1339 std::vector<std::pair<K, boost::optional<V>>> result;
1340 for (size_t i = range.first; i < range.second; ++i) {
1341 auto entry = entries[i];
1342 result.push_back(std::make_pair(entry.first, entry.second));
1343 }
1344 return result;
1345 }
