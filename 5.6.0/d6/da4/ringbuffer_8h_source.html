255 rb::read_result_set get_result_set(boost::future<protocol::ClientMessage> f)
256 {
257 auto msg = f.get();
258 auto* initial_frame =
259 reinterpret_cast<protocol::ClientMessage::frame_header_type*>(
260 msg.rd_ptr(protocol::ClientMessage::RESPONSE_HEADER_LEN));
261 auto read_count = msg.get<int32_t>();
262 auto next_seq = msg.get<int64_t>();
263 msg.rd_ptr(static_cast<int32_t>(initial_frame->frame_len) -
264 protocol::ClientMessage::RESPONSE_HEADER_LEN -
265 protocol::ClientMessage::INT32_SIZE -
266 protocol::ClientMessage::INT64_SIZE);
267
268 auto datas = msg.get<std::vector<serialization::pimpl::data>>();
269 auto item_seqs = msg.get_nullable<std::vector<int64_t>>();
270 return rb::read_result_set(read_count,
271 std::move(datas),
272 get_serialization_service(),
273 item_seqs,
274 next_seq);
275 }
