53 class executor_promise

55 public:
56 executor_promise(spi::ClientContext& context)
57 : context_(context)

61 executor_promise(
62 boost::future<boost::optional<T>>& future,
63 boost::uuids::uuid uuid,
64 int partition_id,
65 boost::uuids::uuid member,
66 spi::ClientContext& context,
67 const std::shared_ptr<spi::impl::ClientInvocation>& invocation)
68 : shared_future_(future.share())
69 , uuid_(uuid)
70 , partition_id_(partition_id)
71 , member_uuid_(member)
72 , context_(context)
73 , invocation_(invocation)
74 {
75 }
76
77 bool cancel(bool may_interrupt_if_running)
78 {
79 if (shared_future_.is_ready()) {
80 return false;
81 }
82
83 try {
84 return invoke_cancel_request(may_interrupt_if_running);
85 } catch (exception::iexception&) {
86 util::exception_util::rethrow(std::current_exception());
87 }
88 return false;
89 }
90
91 boost::shared_future<boost::optional<T>> get_future()
92 {
93 return shared_future_;
94 }
95
96 private:
97 boost::shared_future<boost::optional<T>> shared_future_;
98 boost::uuids::uuid uuid_;
99 int partition_id_;
100 boost::uuids::uuid member_uuid_;
101 spi::ClientContext& context_;
102 std::shared_ptr<spi::impl::ClientInvocation> invocation_;
103
104 bool invoke_cancel_request(bool may_interrupt_if_running)
105 {
106 invocation_->wait_invoked();
107
108 if (partition_id_ > -1) {
109 auto request =
110 protocol::codec::executorservice_cancelonpartition_encode(
111 uuid_, may_interrupt_if_running);
112 std::shared_ptr<spi::impl::ClientInvocation> clientInvocation =
113 spi::impl::ClientInvocation::create(
114 context_,
115 request,
116 boost::uuids::to_string(uuid_),
117 partition_id_);
118 return clientInvocation->invoke()
119 .get()
120 .get_first_fixed_sized_field<bool>();
121 } else {
122 auto request =
123 protocol::codec::executorservice_cancelonmember_encode(
124 uuid_, member_uuid_, may_interrupt_if_running);
125 std::shared_ptr<spi::impl::ClientInvocation> clientInvocation =
126 spi::impl::ClientInvocation::create(
127 context_,
128 request,
129 boost::uuids::to_string(uuid_),
130 member_uuid_);
131 return clientInvocation->invoke()
132 .get()
133 .get_first_fixed_sized_field<bool>();
134 }
135 }

369 template<typename HazelcastSerializable, typename T>
370 executor_promise<T> submit(const HazelcastSerializable& task)
371 {
372 serialization::pimpl::data task_data =
373 to_data<HazelcastSerializable>(task);
374
375 if (task_data.has_partition_hash()) {
376 int partitionId = get_partition_id(task_data);
377
378 return submit_to_partition_internal<T>(
379 task_data, false, partitionId);
380 } else {
381 return submit_to_random_internal<T>(task_data, false);
382 }
383 }

541 template<typename T>
542 typename std::enable_if<!std::is_same<executor_marker, T>::value,
543 executor_promise<T>>::type
544 check_sync(
545 std::pair<boost::future<protocol::ClientMessage>,
546 std::shared_ptr<spi::impl::ClientInvocation>>& future_pair,
547 boost::uuids::uuid uuid,
548 int partition_id,
549 const member& member,
550 bool prevent_sync)
551 {
552 bool sync = is_sync_computation(prevent_sync);
553 auto objectFuture = to_object<T>(
554 decode<serialization::pimpl::data>(std::move(future_pair.first)));
555 if (sync) {
556 objectFuture.wait();
557 }
558
559 return executor_promise<T>(objectFuture,
560 uuid,
561 partition_id,
562 member.get_uuid(),
563 get_context(),
564 future_pair.second);
565 }

580 future_pair.first.get();
581 }
