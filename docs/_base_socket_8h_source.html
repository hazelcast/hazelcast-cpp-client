55  void connect(const std::shared_ptr<connection::Connection> connection) override {
56  using namespace boost::asio;
57  using namespace boost::asio::ip;
58 
59  boost::asio::steady_timer connectTimer(socket_.get_executor());
60  connectTimer.expires_from_now(connect_timeout_);
61  connectTimer.async_wait([=](const boost::system::error_code &ec) {
62  if (ec == boost::asio::error::operation_aborted) {
63  return;
64  }
65  close();
66  return;
67  });
68  try {
69  auto addresses = resolver_.resolve(remote_endpoint_.get_host(), std::to_string(remote_endpoint_.get_port()));
70  boost::asio::async_connect(socket_.lowest_layer(), addresses,
71  boost::asio::use_future).get();
72  post_connect();
73  connectTimer.cancel();
74  set_socket_options(socket_options_);
75  static constexpr const char *PROTOCOL_TYPE_BYTES = "CP2";
76  write(socket_, boost::asio::buffer(PROTOCOL_TYPE_BYTES, 3));
77  } catch (...) {
78  connectTimer.cancel();
79  close();
80  throw;
81  }
82 
83  socket_.lowest_layer().native_non_blocking(true);
84  do_read(std::move(connection));
85  }

87  void async_write(const std::shared_ptr<connection::Connection> connection,
88  const std::shared_ptr<spi::impl::ClientInvocation> invocation) override {
89  check_connection(connection, invocation);
90  auto message = invocation->get_client_message();
91  boost::asio::post(socket_.get_executor(), [=]() {
92  if (!check_connection(connection, invocation)) {
93  return;
94  }
95 
96  bool success;
97  int64_t message_call_id;
98  do {
99  auto call_id = ++call_id_counter_;
100  struct correlation_id {
101  int32_t connnection_id;
102  int32_t call_id;
103  };
104  union {
105  int64_t id;
106  correlation_id composed_id;
107  } c_id_union;
108  c_id_union.composed_id = {connection->get_connection_id(), call_id};
109  message_call_id = c_id_union.id;
110  message->set_correlation_id(c_id_union.id);
111  success = connection->invocations.insert({message_call_id, invocation}).second;
112  } while (!success);
113 
114  auto handler = [=](const boost::system::error_code &ec,
115  std::size_t bytes_written) {
116  if (ec) {
117  auto invocationIt = connection->invocations.find(message_call_id);
118 
119  assert(invocationIt != connection->invocations.end());
120 
121  auto message = (boost::format{
122  "Error %1% during invocation write for %2% on connection %3%"} %
123  ec % *invocation % *connection).str();
124  invocationIt->second->notify_exception(
125  boost::enable_current_exception(
126  std::make_exception_ptr(
127  exception::io(
128  "Connection::write",
129  message))));
130 
131  connection->close(message);
132  connection->invocations.erase(invocationIt);
133  }
134  };
135 
136  auto &datas = message->get_buffer();
137  if (datas.size() == 1) {
138  boost::asio::async_write(socket_, boost::asio::buffer(datas[0]), handler);
139  } else {
140  std::vector<boost::asio::const_buffer> buffers;
141  buffers.reserve(datas.size());
142  for (auto &d : datas) {
143  buffers.push_back(boost::asio::buffer(d));
144  }
145  boost::asio::async_write(socket_, buffers, handler);
146  }
147  });
148  }

200  if (bufferSize > 0) {

209  void do_read(const std::shared_ptr<connection::Connection> connection) {
210  using namespace boost::asio;
211  using namespace boost::asio::ip;
212 
213  socket_.async_read_some(buffer(connection->read_handler.byte_buffer.ix(),
214  connection->read_handler.byte_buffer.remaining()),
215  [=](const boost::system::error_code &ec, std::size_t bytes_read) {
216  if (ec) {
217  // prevent any exceptions
218  util::IOUtil::close_resource(connection.get(),
219  (boost::format("Socket read error. %1% for %2%")
220  %ec %(*connection)).str().c_str());
221  return;
222  }
223 
224  connection->read_handler.byte_buffer.safe_increment_position(
225  bytes_read);
226 
227  connection->read_handler.handle();
228 
229  do_read(connection);
230  });
