234  template<typename E>
235  size_t drain_items(boost::future<std::vector<serialization::pimpl::data>> f, std::vector<E> &elements) {
236  auto datas = f.get();
237  auto size = datas.size();
238  elements.reserve(size);
239  auto &ss = get_context().get_serialization_service();
240  for (auto &data : datas) {
241  elements.push_back(ss.template to_object<E>(data).value());
242  }
243  return size;
244  }
