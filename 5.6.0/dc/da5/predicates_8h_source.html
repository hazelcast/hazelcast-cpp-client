53class multi_predicate : public base_predicate
54{
55public:
56 template<typename... Args>
57 multi_predicate(hazelcast_client& client, const Args&... values)
58 : base_predicate(client)
59 {
60 out_stream.write<int32_t>(static_cast<int32_t>(sizeof...(values)));
61 out_stream.write_objects(values...);
62 }
63
64 template<typename... Args>
65 multi_predicate(const std::string attribute_name,
66 hazelcast_client& client,
67 const Args&... values)
68 : base_predicate(client)
69 {
70 out_stream.write(attribute_name);
71 out_stream.write<int32_t>(static_cast<int32_t>(sizeof...(values)));
72 out_stream.write_objects(values...);
73 }
74
75 template<typename T>
76 multi_predicate(const std::string attribute_name,
77 hazelcast_client& client,
78 const std::vector<T>& values)
79 : base_predicate(client)
80 {
81 out_stream.write(attribute_name);
82 out_stream.write<int32_t>(static_cast<int32_t>(values.size()));
83 for (const T& value : values) {
84 out_stream.write_object(value);
85 }
86 }
87};
